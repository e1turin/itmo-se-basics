\begin{enumerate}
\FunctionalReq{Авторизация}{FR7}{высокий}{Система должна предоставлять возможность редактору авторизироваться в системе.}{20 часов (реализация функциональности, добавление в пользовательский интерфейс, интеграция с базой данных).}{FR2}
\Scenario{Редактор авторизуется в системе}{
    \begin{itemize}
        \item Редактор хочет авторизоваться в системе;
        \item Когда редактор выполняет вход в систему с помощью учетной записи <<Редактора>>,
        \item Тогда редактор должен быть авторизован как <<Редактор>>.
    \end{itemize}
}

\FunctionalReq{Управление новостями}{FR8}{высокий}{Система должна предоставлять возможность редактору создавать новые новости, редактировать их и скрывать из публичного доступа.}{30 часов (реализация функциональности, добавление в пользовательский интерфейс, интеграция с базой данных).}{FR4, FR7}
\Scenario{Создание нового новостного материала}{
    \begin{itemize}
        \item Данный редактор авторизован в системе;
        \item Когда редактор хочет добавить новый новостной материал,
        \item Тогда система должна добавить данный материал в базу данных.
    \end{itemize}
}
\Scenario{Редактирование своего новостного материала}{
    \begin{itemize}
        \item Данный редактор авторизован в системе;
        \item Редактируемый материал находится в базе данных;
        \item Редактируемый материал принадлежит данному редактору;
        \item Когда редактору хочет изменить новостной материал,
        \item Тогда система должна отредактировать данный материал в базе данных.
    \end{itemize}
}
\Scenario{Скрытие новостного материала}{
    \begin{itemize}
        \item Данный редактор авторизован в системе;
        \item Скрываемый материал находится в базе данных;
        \item Скрываемый материал принадлежит данному редактору;
        \item Когда редактор хочет скрыть новостной материал,
        \item Тогда система должна пометить данный материал в базе данных как скрытый.
    \end{itemize}
}
\Scenario{Удаление новостного материала}{
    \begin{itemize}
        \item Данный редактор авторизован в системе;
        \item Удаляемый материал находится в базе данных;
        \item Удаляемый материал принадлежит данному редактору;
        \item Когда редактор хочет удалить новостной материал,
        \item Тогда система должна удалить данный материал из базе данных.
    \end{itemize}
}

\FunctionalReq{Поиск по своим новостям}{FR9}{высокий}{Система должна предоставлять возможность редактору искать новости среди созданных им же.}{20 часов (реализация функциональности, добавление в пользовательский интерфейс, интеграция с базой данных).}{FR5, FR7}
\Scenario{Поиск с фильтрацией среди своих новостных материалов}{
    \begin{itemize}
        \item Данный редактор авторизован в системе;
        \item Редактор вводит фильтр для поиска;
        \item Когда редактор хочет показать отфильтрованные новостные материалы,
        \item Тогда система должна предоставить принадлежащие редактору данные материалы из базы данных.
    \end{itemize}
}
\end{enumerate}
